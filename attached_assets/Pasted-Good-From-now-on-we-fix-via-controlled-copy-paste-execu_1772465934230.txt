router.get("/api/worker/next", async (req, res) => {
  try {
    const auth = req.headers.authorization;

    if (!auth || auth !== `Bearer ${process.env.WORKER_TOKEN}`) {
      return res.status(401).json({ error: "Unauthorized" });
    }

    // Find first approved job
    const approvedJob = await db.query.jobs.findFirst({
      where: (jobs, { eq }) => eq(jobs.status, "approved"),
      orderBy: (jobs, { asc }) => asc(jobs.createdAt),
    });

    if (!approvedJob) {
      return res.json({ job: null });
    }

    // Atomically mark as running
    await db
      .update(jobs)
      .set({
        status: "running",
        startedAt: new Date(),
      })
      .where(eq(jobs.id, approvedJob.id));

    return res.json({
      job: {
        ...approvedJob,
        status: "running",
      },
    });
  } catch (error) {
    console.error("Worker next error:", error);
    return res.status(500).json({ error: "Internal server error" });
  }
});

IMPORTANT:
